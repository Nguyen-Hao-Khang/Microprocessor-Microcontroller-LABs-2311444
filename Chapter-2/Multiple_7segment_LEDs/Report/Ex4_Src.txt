int led_counter = 25;
int dot_counter = 100;
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
{
	if (led_counter > 0)
	{
		led_counter--;
		if (led_counter <= 0)
		{
			led_counter = 25;
			update7SEG(index_led++);
			if (index_led >= MAX_LED) index_led = 0;
		}

	}

	if (dot_counter > 0)
	{
		dot_counter--;
		if (dot_counter <= 0)
		{
			dot_counter = 100;
			HAL_GPIO_TogglePin(LED_RED_GPIO_Port, LED_RED_Pin);
			HAL_GPIO_TogglePin(DOT_GPIO_Port, DOT_Pin);
		}
	}
}
